110 void Game::getMapData(std::vector<std::vector<char>>& map_vector, std::ifstream& map_grid_file)

134 {
135  for (int i = 0; i < map_vector.size(); i++)
136  {
137  auto temp_vector = map_vector[i];
138  for (int j = 0; j < temp_vector.size(); j++)

140  switch(temp_vector[j])

143  createBarrier(j*_game_sprite_dimensions.barrier_sprite_x, i*_game_sprite_dimensions.barrier_sprite_y);

149  addNewTank(p1_tank, _player1_startX, _player1_startY, 0);
